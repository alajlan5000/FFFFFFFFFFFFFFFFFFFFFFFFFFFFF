<script>
const BASE_URL = "https://x8ki-letl-twmt.n7.xano.io/api:Xiw-CLaJ/groups";
let groupsData = [];
let editModeId = null;

// تحميل القروبات والتخصصات
async function fetchGroups() {
try {
const response = await fetch(BASE_URL);
groupsData = await response.json();

const filterSelect = document.getElementById("filterMajor");
const addSelect = document.getElementById("course");

filterSelect.innerHTML = '<option value="">اختر التخصص</option>';
addSelect.innerHTML = '<option value="">اختر التخصص</option>';

let majors = [...new Set(groupsData.map(g => g.Major).filter(m => m && m.trim() !== ""))];

majors.forEach(m => {
const opt1 = document.createElement("option");
opt1.value = m;
opt1.textContent = m;
filterSelect.appendChild(opt1);

const opt2 = document.createElement("option");
opt2.value = m;
opt2.textContent = m;
addSelect.appendChild(opt2);
});

filterGroups();
} catch (err) {
console.error("خطأ في جلب القروبات:", err);
}
}

// عرض وتصفية القروبات
function filterGroups() {
const selectedMajor = document.getElementById("filterMajor").value.trim();
const container = document.getElementById("groupsList");
container.innerHTML = "";

const filtered = groupsData.filter(g => !selectedMajor || g.Major === selectedMajor);

if (filtered.length > 0) {
filtered.forEach(g => {
container.innerHTML += `
<div class="group-card">
<h3>${g.Course}</h3>
<p>الشعبة: ${g.Section}</p>
<p>التخصص: ${g.Major}</p>
<a href="${g.Link}" target="_blank">انضم</a>
<button class="edit-btn" onclick='editGroup(${g.id})'>تعديل</button>
</div>
`;
});
} else {
container.innerHTML = "<p class='empty'>❌ لا يوجد قروبات مطابقة للتخصص</p>";
}
}

// إضافة أو تعديل قروب
async function addGroup(event) {
event.preventDefault();
const Major = document.getElementById("course").value.trim();
const Section = document.getElementById("section").value.trim();
const Course = document.getElementById("subject").value.trim();
const Link = document.getElementById("link").value.trim();

if (!Major || !Section || !Course || !Link) {
alert("اكمل جميع الحقول");
return;
}

const body = { Major, Section, Course, Link };

try {
let result;
if (editModeId !== null) {
// PATCH لتحديث السجل
const res = await fetch(`${BASE_URL}/${editModeId}`, {
method: "PATCH",
headers: { "Content-Type": "application/json" },
body: JSON.stringify(body)
});
result = await res.json();
const idx = groupsData.findIndex(g => g.id === editModeId);
if (idx >= 0) groupsData[idx] = result;
editModeId = null;
document.getElementById("submitBtn").innerText = "إرسال ✅";
alert("✅ تم التعديل بنجاح");
} else {
// POST لإضافة سجل جديد
const res = await fetch(BASE_URL, {
method: "POST",
headers: { "Content-Type": "application/json" },
body: JSON.stringify(body)
});
result = await res.json();
groupsData.push(result);
alert("✅ تمت الإضافة بنجاح");
}

document.getElementById("addForm").reset();
fetchGroups();
} catch (err) {
console.error("خطأ أثناء العملية:", err);
alert("حدث خطأ أثناء العملية");
}
}

// تفعيل وضع التعديل
function editGroup(id) {
const group = groupsData.find(g => g.id === id);
if (!group) return;

document.getElementById("course").value = group.Major;
document.getElementById("section").value = group.Section;
document.getElementById("subject").value = group.Course;
document.getElementById("link").value = group.Link;
editModeId = group.id;
document.getElementById("submitBtn").innerText = "تعديل ✏️";
window.scrollTo(0, 0);
}

fetchGroups();
</script>
